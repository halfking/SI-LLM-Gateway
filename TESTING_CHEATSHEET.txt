💾 数据库查询
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
# 设置数据库连接
export LLM_GATEWAY_DATABASE_URL="postgresql://user:pass@host:port/db"

# 快速统计
psql "$LLM_GATEWAY_DATABASE_URL" -c "
SELECT 
    COUNT(*) as total,
    COUNT(*) FILTER (WHERE success) as success,
    COUNT(*) FILTER (WHERE error_kind = 'empty_response') as empty_resp
FROM request_logs
WHERE client_model = 'minimax-m3' AND ts > NOW() - INTERVAL '1 hour';"

# 按凭据统计
psql "$LLM_GATEWAY_DATABASE_URL" -c "
SELECT c.id, c.label, COUNT(*) as requests,
       COUNT(*) FILTER (WHERE rl.success) as success
FROM request_logs rl
JOIN credentials c ON c.id = rl.credential_id
WHERE rl.client_model = 'minimax-m3' AND rl.ts > NOW() - INTERVAL '1 hour'
GROUP BY c.id, c.label ORDER BY requests DESC;"

🎯 问题排查流程
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
1. 运行基础测试
   → ./scripts/test_71_complete.sh

2. 如果出现 "no candidate" 错误
   → ./scripts/diagnose_routing_issue.sh minimax-m3

3. 如果出现 "empty response" 错误
   → ./scripts/diagnose_nvidia_nim_empty_response.sh minimax-m3

4. 查看详细日志
   → cat test_logs_71/responses_*.log

5. 检查服务端日志
   → tail -f /var/log/llm-gateway/app.log
   → grep "request_id" /var/log/llm-gateway/*.log
