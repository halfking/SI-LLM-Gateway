快速命令参考
================================================================================

# 检查数据库状态
psql -d llm_gateway -c "\dt"

# 检查视图
psql -d llm_gateway -c "\dv v_routable_credential_models"

# 查看可路由节点
psql -d llm_gateway -c "
SELECT COUNT(*) as routable_nodes
FROM v_routable_credential_models 
WHERE tenant_id = 'default' AND is_routable = TRUE;
"

# 运行完整验证
psql -d llm_gateway -c "
SELECT 
    COUNT(*) as 总数,
    COUNT(*) FILTER (WHERE is_routable) as 可路由,
    COUNT(*) FILTER (WHERE NOT is_routable) as 不可路由
FROM v_routable_credential_models
WHERE tenant_id = 'default';
"

# 重新运行修复（如果需要）
./diagnose_and_fix.sh

# 或手动运行
